CREATE TABLE `bucket_item` (
  `idx` int NOT NULL AUTO_INCREMENT,
  `bucket_list_idx` int DEFAULT NULL,
  `title` varchar(64) DEFAULT NULL,
  `content` varchar(256) DEFAULT NULL,
  `completion_date` datetime DEFAULT NULL,
  `member_idx` int DEFAULT NULL,
  PRIMARY KEY (`idx`),
  KEY `bucket_list_idx` (`bucket_list_idx`),
  KEY `member_idx` (`member_idx`),
  CONSTRAINT `bucket_item_ibfk_1` FOREIGN KEY (`bucket_list_idx`) REFERENCES `bucket_list` (`idx`),
  CONSTRAINT `bucket_item_ibfk_2` FOREIGN KEY (`member_idx`) REFERENCES `member` (`idx`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
